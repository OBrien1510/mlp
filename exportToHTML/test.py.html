from MultiLayerPerceptron import MultiLayerPerceptron
import numpy as np

mlp = MultiLayerPerceptron((2, 2, 1), hidden_activation="sigmoid", max_iters=10000, linear_factor=0.1, learning_rate=0.5,
                           verbose=(True, 1000,), output_activation="linear")

starting_weights = mlp.weights

x = np.array([[1, 0], [0, 1], [0, 0], [1, 1]])
y = np.array([1, 1, 0, 0])

mlp.fit(x, y)

print("1, 1")
print(mlp.predict(np.array([1, 1])))
print("0, 0")
print(mlp.predict(np.array([0, 0])))
print("0, 1")
print(mlp.predict(np.array([0, 1])))
print("1, 0")
print(mlp.predict(np.array([1, 0])))

mlp.to_str()
